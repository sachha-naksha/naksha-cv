% Section label hanging in the left margin: small caps, purple, top-aligned with
% the first content line, left-justified (wraps within \cvlabelw if long).
% The label is wrapped in \raisebox{0pt}[\height][0pt]{...} so it keeps its
% HEIGHT (won't crash into the section above) but reports ZERO DEPTH. Without
% this, a label that wraps to 2-3 lines (e.g. "Leadership & Societal
% Contributions") pushes the first content line down by its full height, leaving
% a huge gap before the content when the heading itself is short. Zero depth lets
% the content flow normally beside the label, which simply extends into the margin.
\newcommand{\cvsection}[1]{%
  \par\addvspace{1.4ex}\needspace{3\baselineskip}%
  \noindent\llap{\raisebox{0pt}[\height][0pt]{%
    \parbox[t]{\cvlabelw}{\raggedright\normalfont\scshape\bfseries
      \textcolor{sectionpurple}{#1}}}\hspace{\cvlabelsep}}%
  \nopagebreak\ignorespaces}

% One entry: left content (wraps freely) with the date pinned to the top-right.
% #1 may contain \\ for additional detail lines under the title.
\newcommand{\entry}[2]{%
  \parbox[t]{\dimexpr\linewidth-\cvdatew-\cvlabelsep\relax}{\raggedright #1}%
  \hfill
  \parbox[t]{\cvdatew}{\raggedleft\small\textsl{#2}}\par}

% Right-flushed italic metadata for use INSIDE bullet items (short text only).
\newcommand{\dates}[1]{\hfill{\small\textsl{#1}}}

\setlist[itemize]{leftmargin=1.3em, itemsep=0pt, parsep=1pt, topsep=2pt, partopsep=0pt}

% Bullet list for experience-style entries (heading + points). Use it in place
% of itemize right after an \entry. The spacing is STRUCTURAL, so cutting bullet
% text can never reopen the gap bug:
%   before=\nointerlineskip  cancels the variable baseline glue that \entry's
%     \parbox[t] injects (1-line headings sat far from bullets, 2-line ones on
%     top of them) -> heading always binds tightly, no marker needed.
%   after=\addvspace{7pt}    gives an even separation before the next entry.
\newlist{roleitems}{itemize}{1}
\setlist[roleitems]{label=$\bullet$, leftmargin=1.3em, itemsep=1pt, parsep=0pt,
  topsep=2pt, partopsep=0pt, before=\nointerlineskip, after=\addvspace{7pt}}
